local Physics = {}

-- Collision detection function
function Physics.checkCollision(ball1, ball2)
    local dx = ball1.x - ball2.x
    local dy = ball1.y - ball2.y
    local distance = math.sqrt(dx * dx + dy * dy)
    return distance < (ball1.radius + ball2.radius)
end

-- Check collision between ball and fire
function Physics.checkFireCollision(ball, fire)
    local dx = ball.x - fire.x
    local dy = ball.y - fire.y
    local distance = math.sqrt(dx * dx + dy * dy)
    return distance < (ball.radius + fire.radius)
end

-- Collision response function
function Physics.resolveCollision(ball1, ball2)
    local dx = ball1.x - ball2.x
    local dy = ball1.y - ball2.y
    local distance = math.sqrt(dx * dx + dy * dy)
    
    if distance == 0 then return end  -- Prevent division by zero
    
    -- Normalize the collision vector
    local nx = dx / distance
    local ny = dy / distance
    
    -- Calculate overlap
    local overlap = (ball1.radius + ball2.radius) - distance
    
    -- Separate the balls
    local separationX = nx * overlap * 0.5
    local separationY = ny * overlap * 0.5
    
    ball1.x = ball1.x + separationX
    ball1.y = ball1.y + separationY
    ball2.x = ball2.x - separationX
    ball2.y = ball2.y - separationY
    
    -- Apply velocity changes (simple elastic collision)
    local relativeVelocityX = ball1.vx - ball2.vx
    local relativeVelocityY = ball1.vy - ball2.vy
    
    local velocityAlongNormal = relativeVelocityX * nx + relativeVelocityY * ny
    
    if velocityAlongNormal > 0 then return end  -- Objects separating
    
    local restitution = 0.8  -- Bounciness factor
    local impulse = -(1 + restitution) * velocityAlongNormal / 2
    
    ball1.vx = ball1.vx + impulse * nx
    ball1.vy = ball1.vy + impulse * ny
    ball2.vx = ball2.vx - impulse * nx
    ball2.vy = ball2.vy - impulse * ny
end

return Physics
