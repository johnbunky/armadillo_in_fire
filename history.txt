-- Generate a simple tone sound programmatically
function Audio:generateTone(frequency, duration, volume)
    local sampleRate = 44100
    local samples = sampleRate * duration
    local soundData = love.sound.newSoundData(samples, sampleRate, 16, 1)
    
    for i = 0, samples - 1 do
        local t = i / sampleRate
        local wave = math.sin(2 * math.pi * frequency * t) 
        -- Add envelope to prevent clicking
        local envelope = 1
        if t < 0.1 then
            envelope = t / 0.1
        elseif t > duration - 0.1 then
            envelope = (duration - t) / 0.1
        end
        soundData:setSample(i, wave * volume * envelope)
    end
    
    return love.audio.newSource(soundData, "static")
end

-- Generate crackling fire sound using noise
function Audio:generateCrackling(duration, volume)
    local sampleRate = 44100
    local samples = sampleRate * duration
